## 添加远程仓库地址
git remote set-url --add origin [url]

### 修改远程仓库地址
git remote set-url origin [url]

### 删除远程仓库地址
git remote rm origin

### 新建远程仓库地址
git remote add origin [url]

### 查看远程仓库地址
git remote -v
